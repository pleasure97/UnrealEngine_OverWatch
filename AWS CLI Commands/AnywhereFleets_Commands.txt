5. Create Game Session 
aws gamelift create-game-session ^
 --fleet-id fleet-51a5e70e-2ad6-49bd-ad47-16ade7a32b51 ^
 --name AnywhereFleetSession ^

 --region ap-northeast-2 ^
 --location custom-home-desktop

6. Find Game Session 
aws gamelift describe-game-sessions ^
 --fleet-id fleetId* ^
 --status-filter ACTIVE

7. Create Player Session
aws gamelift create-player-session ^
 --game-session-id gameSessionId(acquired by describe-game-sessions command) ^
 --player-id playerName*
